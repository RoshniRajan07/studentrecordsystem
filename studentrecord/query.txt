CREATE TABLE STUDENT_TB(
STUDENTID VARCHAR2(12) PRIMARY KEY,
STUDENTNAME VARCHAR2(50) NOT NULL,
ROLLNUMBER VARCHAR2(20) NOT NULL,
CLASSNAME VARCHAR2(10) NOT NULL,
ADMISSION_DATE DATE ,
AGE NUMBER ,
REMARKS VARCHAR2(100)
);
CREATE SEQUENCE STUDENT_SEQ1
START WITH 10
INCREMENT BY 1
MAXVALUE 99;
